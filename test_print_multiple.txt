// Test printing multiple expressions

int x = 10;
int y = 20;
int z = 30;

// Print single value
print x;

// Print multiple values
print x, y, z;

// Print expressions
print x + y, x * 2, z - 5;

// Print mixed
print x, x + y, y, y * z;

// In a loop
loop from i = 0 to 2 {
    print i, i * 2, i + 10;
}
