// Test return statement checking

// Error: int function without return
func int noReturn(int x) {
    print x;
}

// OK: int function with return
func int withReturn(int x) {
    return x + 1;
}

// OK: void function without return
func void voidFunc(int x) {
    print x;
}

// OK: void function with empty return
func void voidFunc2(int x) {
    print x;
    return;
}

// Error: int function with conditional return (not all paths)
func int conditionalReturn(int x) {
    if (x > 0) {
        return x;
    }
    // Missing return for else case
}

// OK: int function with return in all paths
func int allPathsReturn(int x) {
    if (x > 0) {
        return x;
    } else {
        return 0;
    }
}

int a = withReturn(5);
print a;
