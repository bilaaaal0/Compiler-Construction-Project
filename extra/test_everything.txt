// Recursive factorial
func int factorial(int n) {
    if (n <= 1) {
        return 1;
    }
    int temp;
    temp = n - 1;
    return n * factorial(temp);
}

// Check if prime (with loop and if-else)
func int isPrime(int n) {
    if (n <= 1) {
        return 0;
    }
    if (n == 2) {
        return 1;
    }
    
    int result;
    result = 1;
    
    loop from i = 2 to n {
        int remainder;
        remainder = n % i;
        if (remainder == 0) {
            if (i != n) {
                result = 0;
            }
        }
    }
    
    return result;
}
