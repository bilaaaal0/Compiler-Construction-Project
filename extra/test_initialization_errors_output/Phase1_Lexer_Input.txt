// Test initialization errors

// Error 1: Using uninitialized variable in expression
int x;
int y = x + 5;  // ERROR: x not initialized

// Error 2: Using uninitialized variable in loop
int i;
loop from i to 5 {  // ERROR: i not initialized
    print i;
}

// Error 3: Printing uninitialized variable
int z;
print z;  // ERROR: z not initialized
