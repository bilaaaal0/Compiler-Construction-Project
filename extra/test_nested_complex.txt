// Test nested loops, nested if, and complex conditions

int x = 5;
int y = 10;
int z = 15;

// Test complex logical expression
if ((x == 5 && y == 10) || z == 20) {
    print x;
    print y;
}

// Test nested if statements
if (x > 0) {
    print x;
    if (y > 5) {
        print y;
        if (z > 10) {
            print z;
        }
    }
}

// Test nested loops
int i = 0;
int j = 0;

loop from i = 0 to 3 {
    print i;
    loop from j = 0 to 2 {
        print j;
    }
}

// Test complex nested structure
int a = 1;
int b = 2;

loop from a = 0 to 2 {
    if (a == 0 || a == 2) {
        print a;
        loop from b = 0 to 2 {
            if ((b > 0 && a > 0) || b == 0) {
                print b;
            }
        }
    }
}
