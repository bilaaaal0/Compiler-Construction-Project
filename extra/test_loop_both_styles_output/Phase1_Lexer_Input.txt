// Test both loop styles
int sum1;
int sum2;

sum1 = 0;
sum2 = 0;

// Style 1: Auto-declare variable in loop
loop from i = 1 to 10 {
    sum1 = sum1 + i;
}

print sum1;

// Style 2: Use pre-declared variable
int j;
j = 5;
loop from j to 15 {
    sum2 = sum2 + j;
}

print sum2;

// Style 3: With step and pre-declared variable
int k;
k = 0;
loop from k to 20 step 2 {
    print k;
}
