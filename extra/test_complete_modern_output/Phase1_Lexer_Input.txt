// Function: Calculate factorial
func int factorial(int n) {
    if (n <= 1) {
        return 1;
    }
    int temp;
    temp = n - 1;
    return n * factorial(temp);
}

// Function: Check if even
func int isEven(int x) {
    int remainder;
    remainder = x % 2;
    if (remainder == 0) {
        return 1;
    }
    return 0;
}
