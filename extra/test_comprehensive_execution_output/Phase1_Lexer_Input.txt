// Function 1: Recursive factorial
func int factorial(int n) {
    if (n <= 1) {
        return 1;
    }
    int prev;
    prev = n - 1;
    int result;
    result = factorial(prev);
    return n * result;
}

// Function 2: Recursive fibonacci
func int fibonacci(int n) {
    if (n <= 0) {
        return 0;
    }
    if (n == 1) {
        return 1;
    }
    int a;
    int b;
    a = n - 1;
    b = n - 2;
    return fibonacci(a) + fibonacci(b);
}

// Function 3: Check if even (with if-else)
func int isEven(int x) {
    int remainder;
    remainder = x % 2;
    if (remainder == 0) {
        return 1;
    } else {
        return 0;
    }
}

// Function 4: Get grade (nested if-elif-else)
func int getGrade(int score) {
    if (score >= 90) {
        return 1;
    } elif (score >= 80) {
        return 2;
    } elif (score >= 70) {
        return 3;
    } else {
        return 0;
    }
}
