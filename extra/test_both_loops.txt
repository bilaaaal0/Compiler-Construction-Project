// Test: Both old and new loop syntax work together!

int sum1;
int sum2;

sum1 = 0;
sum2 = 0;

// Old C-style for loop
for (i = 1; i <= 5; i = i + 1) {
    sum1 = sum1 + i;
}

// New modern loop syntax
loop from j = 1 to 5 {
    sum2 = sum2 + j;
}

// Both should give same result
print sum1;  // 15
print sum2;  // 15

// New syntax with step
int evens;
evens = 0;

loop from k = 0 to 10 step 2 {
    evens = evens + k;
}

print evens;  // 30 (0+2+4+6+8+10)
