// Comprehensive scope test

int x = 10;
int y = 20;

print x;  // Should print 10
print y;  // Should print 20

{
    x = 100;  // Modifies x in this scope
    int z = 30;  // New variable in this scope
    print x;  // Should print 100
    print z;  // Should print 30
    
    {
        x = 200;  // Modifies x in nested scope
        print x;  // Should print 200
    }
    
    print x;  // Should print 100 (restored from nested scope)
}

print x;  // Should print 10 (restored from outer scope)

// Test with if statement
if (y == 20) {
    int a = 50;
    print a;  // Should print 50
}

// Test with loop
int i = 0;
loop from i = 0 to 2 {
    int b = 99;
    print b;  // Should print 99 three times
}
